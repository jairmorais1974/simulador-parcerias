    <script>
        // Seletores
        const inLicencas = document.getElementById('input-licencas');
        const inPreco = document.getElementById('input-preco');
        const inFixoD = document.getElementById('input-fixo-d');
        const inTaxaPJ = document.getElementById('input-taxa-pj');
        const inCustoP = document.getElementById('input-custo-p');
        const inCustoD = document.getElementById('input-custo-d');

        const labLicencas = document.getElementById('label-licencas');
        const labPreco = document.getElementById('label-preco');
        const labFixoD = document.getElementById('label-fixo-d');
        const resBrutoTotal = document.getElementById('res-bruto-total');

        let chart;

        function formatBRL(val) {
            return val.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL', maximumFractionDigits: 0 });
        }

        // Tabela IRPF 2024 Simplificada
        function calcIRPF(mensal) {
            if (mensal <= 2259) return 0;
            if (mensal <= 2826) return (mensal * 0.075) - 169;
            if (mensal <= 3751) return (mensal * 0.15) - 381;
            if (mensal <= 4664) return (mensal * 0.225) - 662;
            return (mensal * 0.275) - 896;
        }

        function update() {
            const licencas = parseInt(inLicencas.value);
            const preco = parseInt(inPreco.value);
            const valorFixoD = parseInt(inFixoD.value);
            const impostoPJ = parseFloat(inTaxaPJ.value) / 100;
            const custoOpP = parseFloat(inCustoP.value) || 0;
            const custoOpD = parseFloat(inCustoD.value) || 0;

            labLicencas.innerText = licencas;
            labPreco.innerText = preco.toLocaleString('pt-BR');
            labFixoD.innerText = valorFixoD.toLocaleString('pt-BR');

            const brutoTotalGeral = licencas * preco;
            resBrutoTotal.innerText = formatBRL(brutoTotalGeral);

            const scenarios = [
                { id: 1, pShare: 0.7, dShare: 0.3, type: 'pct' },
                { id: 2, pShare: 0.5, dShare: 0.5, type: 'pct' },
                { id: 3, dFixed: valorFixoD, type: 'fixed' }
            ];

            const chartDataP = [];
            const chartDataD = [];

            scenarios.forEach(s => {
                let dBrutoAno, pBrutoAno;

                if (s.type === 'fixed') {
                    dBrutoAno = licencas * s.dFixed;
                    pBrutoAno = brutoTotalGeral - dBrutoAno;
                } else {
                    dBrutoAno = brutoTotalGeral * s.dShare;
                    pBrutoAno = brutoTotalGeral * s.pShare;
                }

                // Líquido Parceiro (Considerando Imposto sobre faturamento total)
                const pImpostoTotal = brutoTotalGeral * impostoPJ;
                const pLiqAno = pBrutoAno - pImpostoTotal - (custoOpP * 12);
                
                // Líquido Jair (PF)
                const dBrutoMes = dBrutoAno / 12;
                const dIRMes = calcIRPF(dBrutoMes);
                const dLiqAno = (dBrutoMes - dIRMes - custoOpD) * 12;

                // Atualizar UI
                document.getElementById(`c${s.id}-p-bruto`).innerText = formatBRL(pBrutoAno);
                document.getElementById(`c${s.id}-p-liq`).innerText = formatBRL(Math.max(0, pLiqAno));
                document.getElementById(`c${s.id}-d-bruto`).innerText = formatBRL(dBrutoAno);
                document.getElementById(`c${s.id}-d-liq`).innerText = formatBRL(Math.max(0, dLiqAno));

                chartDataP.push(Math.max(0, pLiqAno));
                chartDataD.push(Math.max(0, dLiqAno));
            });

            updateChart(chartDataP, chartDataD);
        }

        function updateChart(dataP, dataD) {
            const ctx = document.getElementById('chartResults').getContext('2d');
            
            const data = {
                labels: ['Cenário 1 (30/70)', 'Cenário 2 (50/50)', 'Cenário 3 (Fixo)'],
                datasets: [
                    {
                        label: 'Líquido Anual Parceiro',
                        data: dataP,
                        backgroundColor: '#0f172a',
                        borderRadius: 8,
                    },
                    {
                        label: 'Líquido Anual Jair (Dev)',
                        data: dataD,
                        backgroundColor: '#3b82f6',
                        borderRadius: 8,
                    }
                ]
            };

            if (chart) {
                chart.data = data;
                chart.update();
            } else {
                chart = new Chart(ctx, {
                    type: 'bar',
                    data: data,
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        scales: {
                            y: {
                                beginAtZero: true,
                                ticks: {
                                    callback: (v) => 'R$ ' + (v/1000) + 'k'
                                }
                            }
                        },
                        plugins: {
                            legend: {
                                position: 'bottom',
                                labels: { font: { weight: 'bold' }, padding: 20 }
                            },
                            tooltip: {
                                callbacks: {
                                    label: (context) => `${context.dataset.label}: ${formatBRL(context.raw)}`
                                }
                            }
                        }
                    }
                });
            }
        }

        // Listeners
        [inLicencas, inPreco, inFixoD, inTaxaPJ, inCustoP, inCustoD].forEach(el => {
            el.addEventListener('input', update);
        });

        // Inicialização
        window.onload = update;
    </script>
